## 三、轮询工具

```typescript
export function poll<T>(
  fn: () => Promise<T>,
  check: (data: T) => boolean,
  interval = 2500,
  maxAttempts = 60
): Promise<T> {
  return new Promise((resolve, reject) => {
    let attempts = 0
    const timer = setInterval(async () => {
      attempts++
      try {
        const data = await fn()
        if (check(data)) {
          clearInterval(timer)
          resolve(data)
        } else if (attempts >= maxAttempts) {
          clearInterval(timer)
          reject(new Error('处理超时，请稍后重试'))
        }
      } catch (e) {
        clearInterval(timer)
        reject(e)
      }
    }, interval)
  })
}
```

### 使用示例

```typescript
// 轮询识别结果
await poll(
  () => getRecognition(examId),
  (data) => data.data.status === 'recognized'
)

// 轮询分析结果
await poll(
  () => getAnalysis(examId),
  (data) => data.data.status === 'analyzed'
)

// 轮询题目
await poll(
  () => getPracticeQuestions(sessionId),
  (data) => data.data.status === 'ready'
)

// 轮询评估
await poll(
  () => getAssessment(sessionId),
  (data) => data.data.status === 'assessed'
)
```
